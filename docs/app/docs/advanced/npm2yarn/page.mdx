import { compileMdx } from 'nextra/compile'
import { Tabs } from 'nextra/components'
import { MDXRemote } from 'nextra/mdx-remote'

# Npm2Yarn

Nextra 使用
[`@theguild/remark-npm2yarn`](https://npmjs.com/package/@thetheguild/remark-npm2yarn)
包，该包会将带有 `npm2yarn` 元数据的代码块替换为
`nextra/components` 中的
[`<Tabs>` 和 `<Tabs.Tab>` 组件](/docs/built-ins/tabs)。

选中的标签页会保存在本地存储中，在后续页面渲染时将自动选中。

## 示例

<Page />

export async function Page() {
  const codeBlock = `\`\`\`sh npm2yarn
npm i -D @graphql-eslint/eslint-plugin
\`\`\``
  const rawJs = await compileMdx(`${codeBlock}
## Usage
~~~md filename="Markdown" /npm2yarn/
${codeBlock}
~~~`)
  return <MDXRemote compiledSource={rawJs} components={{ $Tabs: Tabs }} />
}
